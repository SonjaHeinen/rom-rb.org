Relationslearn more

class UserRelation < ROM::Relation[:sql]
  dataset :users

  def by_name(name)
    where(name: name)
  end
end

Mapperslearn more

class UserEntityMapper < ROM::Mapper
  relation :users
  register_as :entity
  model User
  attribute :id
  attribute :name
end

Commandslearn more

class CreateUser < ROM::Commands::Create[:sql]
  relation :users
  register_as :create
  input NewUserInput
  validator NewUserValidator
  result :one
end
